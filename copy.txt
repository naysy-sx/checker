    function findLongestCommonSubstring(str1, str2) {
      let longest = '';
      const s1 = str1.toLowerCase();
      const s2 = str2.toLowerCase();

      for (let i = 0; i < s1.length; i++) {
        for (let j = 0; j < s2.length; j++) {
          let k = 0;
          while ((i + k) < s1.length && (j + k) < s2.length &&
            s1[i + k] === s2[j + k]) {
            k++;
          }
          if (k > longest.length) {
            longest = s1.substr(i, k);
          }
        }
      }
      return longest;
    }

    function ratcliffObershelp(s1, s2) {
      if (s1 === s2) return 1.0;
      if (s1.length === 0 || s2.length === 0) return 0.0;

      const lcs = findLongestCommonSubstring(s1, s2);
      if (lcs.length === 0) return 0.0;

      const s1Left = s1.substr(0, s1.indexOf(lcs));
      const s1Right = s1.substr(s1.indexOf(lcs) + lcs.length);
      const s2Left = s2.substr(0, s2.indexOf(lcs));
      const s2Right = s2.substr(s2.indexOf(lcs) + lcs.length);

      return (2.0 * lcs.length + ratcliffObershelp(s1Left, s2Left) +
        ratcliffObershelp(s1Right, s2Right)) / (s1.length + s2.length);
    }
